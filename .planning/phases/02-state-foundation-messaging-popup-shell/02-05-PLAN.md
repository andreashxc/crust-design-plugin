    Step A — Create `apps/extension/src/popup/store.ts` per RESEARCH §8 verbatim (5-slice composition typed via StateCreator):

    ```ts
    /**
     * Zustand store for the popup. D-27.
     *
     * Three slices: registry (registry.json data), storage (mirrored
     * chrome.storage.local state), tab (active tab + applied:<tabId>).
     *
     * useStore.setState(partial) is callable from outside React — used by the
     * chrome.storage.onChanged listener registered in entrypoints/popup/main.tsx
     * (Plan 02-06).
     */
    import { create, type StateCreator } from 'zustand';
    import type {
      AutoDisableRecord,
      ErrorRecord,
      Registry,
    } from '@platform/experiment-sdk';

    // ----- Slice 1: Registry -----
    type RegistrySlice = {
      registry: Registry;
      setRegistry: (r: Registry) => void;
    };
    const registrySlice: StateCreator<RootState, [], [], RegistrySlice> = (set) => ({
      registry: [],
      setRegistry: (r) => set({ registry: r }),
    });

    // ----- Slice 2: Storage-mirrored state -----
    type StorageSlice = {
      enabled: Record<string, boolean>;
      autodisabled: Record<string, AutoDisableRecord>;
      lastError: Record<string, ErrorRecord>;
      setEnabled: (e: Record<string, boolean>) => void;
      setAutoDisabled: (a: Record<string, AutoDisableRecord>) => void;
      setLastError: (m: Record<string, ErrorRecord>) => void;
    };
    const storageSlice: StateCreator<RootState, [], [], StorageSlice> = (set) => ({
      enabled: {},
      autodisabled: {},
      lastError: {},
      setEnabled: (e) => set({ enabled: e }),
      setAutoDisabled: (a) => set({ autodisabled: a }),
      setLastError: (m) => set({ lastError: m }),
    });

    // ----- Slice 3: Active tab + applied set -----
    // Blocker 4 fix (checker iteration 1): activeTabUrl added so computeStatus
    // can resolve 'pending' vs 'applied' per D-26 (CONTEXT) — full active-tab
    // URL is needed to evaluate scope matching against the live page.
    type TabSlice = {
      activeTabId: number | null;
      activeTabUrl: string | null;
      appliedInActiveTab: string[];
      setActiveTab: (id: number | null) => void;
      setActiveTabUrl: (url: string | null) => void;
      setAppliedInActiveTab: (ids: string[]) => void;
    };
    const tabSlice: StateCreator<RootState, [], [], TabSlice> = (set) => ({
      activeTabId: null,
      activeTabUrl: null,
      appliedInActiveTab: [],
      setActiveTab: (id) => set({ activeTabId: id }),
      setActiveTabUrl: (url) => set({ activeTabUrl: url }),
      setAppliedInActiveTab: (ids) => set({ appliedInActiveTab: ids }),
    });

    // ----- Root composition -----
    export type RootState = RegistrySlice & StorageSlice & TabSlice;

    export const useStore = create<RootState>()((...a) => ({
      ...registrySlice(...a),
      ...storageSlice(...a),
      ...tabSlice(...a),
    }));
    ```

    ```ts
    /**
     * Pure function: derives ExperimentStatus from registry entry + ctx.
     * D-08 (5 states) + D-10 (computed in popup, NOT stored).
     *
     * Status decision tree (in order):
     *  1. autodisabled present → 'auto-disabled'
     *  2. !enabled → 'disabled'
     *  3. !scope matches active tab → 'pending'
     *  4. id in appliedInActiveTab → 'applied'
     *  5. else (enabled + scope + not yet applied) → 'error'
     *     (last apply failed; the popup row separately reads lastError[id]
     *     to render the message and decide between rendering 'error' badge or
     *     a transient pending state — see UI-SPEC line 287 for the
     *     bootstrap-loading transient.)
     */
    import type {
      AutoDisableRecord,
      ExperimentStatus,
      RegistryEntry,
    } from '@platform/experiment-sdk';
    import { matchesUrl } from '@/shared/url-match';

    export function computeStatus(
      entry: RegistryEntry,
      ctx: {
        enabled: boolean;
        autodisabled?: AutoDisableRecord;
        appliedInActiveTab: string[];
        activeTabUrl?: string;
      },
    ): ExperimentStatus {
      if (ctx.autodisabled) return 'auto-disabled';
      if (!ctx.enabled) return 'disabled';
      const scopeMatches =
        ctx.activeTabUrl != null && matchesUrl(ctx.activeTabUrl, entry.scope.match);
      if (!scopeMatches) return 'pending';
      if (ctx.appliedInActiveTab.includes(entry.id)) return 'applied';
      return 'error';
    }
    ```

    ```ts
    /**
     * Pure function: groups registry entries by author, sorted alphabetical
     * case-insensitive. defaultOpen=true if author owns ≥1 entry whose scope
     * matches the active tab URL.
     *
     * D-22 verbatim semantics:
     *  - Authors sorted alphabetically (case-insensitive).
     *  - Each group shows {author, count = total experiments by author (NOT
     *    just enabled)}.
     *  - defaultOpen if author owns ≥1 experiment matching active tab; else
     *    closed.
     */
    import type { Registry, RegistryEntry } from '@platform/experiment-sdk';

    export type AuthorGroup = {
      author: string;
      entries: RegistryEntry[];
      defaultOpen: boolean;
    };

    export function groupByAuthor(
      registry: Registry,
      ctx: {
        activeTabUrl?: string;
        matchesUrl: (url: string, patterns: string[]) => boolean;
      },
    ): AuthorGroup[] {
      const byAuthor = new Map<string, RegistryEntry[]>();
      for (const e of registry) {
        const list = byAuthor.get(e.author) ?? [];
        list.push(e);
        byAuthor.set(e.author, list);
      }
      return Array.from(byAuthor.entries())
        .sort(([a], [b]) => a.toLowerCase().localeCompare(b.toLowerCase()))
        .map(([author, entries]) => ({
          author,
          entries,
          defaultOpen:
            ctx.activeTabUrl != null
              ? entries.some((e) => ctx.matchesUrl(ctx.activeTabUrl!, e.scope.match))
              : false,
        }));
    }
    ```
