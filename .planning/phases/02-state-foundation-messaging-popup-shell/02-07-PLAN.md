    ```ts
    /**
     * D-02: signal.aborted MUST be true at the time cleanup() runs (controller
     * is aborted BEFORE cleanup() is invoked).
     * RESEARCH §10: AbortError thrown from cleanup is treated as success — NOT
     * escalated via sendMessage('EXPERIMENT_ERROR').
     *
     * Test strategy: model the controller-aborted-before-cleanup contract as a
     * tiny unit harness that mimics the reconcile shape (Task 2's production
     * code mirrors this harness verbatim).
     */
    import { beforeEach, describe, expect, it, vi } from 'vitest';
    import * as messages from '@/shared/messages';

    /**
     * Minimal harness mimicking the reconcile cleanup branch:
     *  controller.abort() then cleanup(); on AbortError swallow; on other Error
     *  call sendMessage('EXPERIMENT_ERROR').
     */
    async function runCleanup(
      cleanup: () => void | Promise<void>,
      controller: AbortController,
      id: string,
    ): Promise<void> {
      controller.abort();
      try {
        await cleanup();
      } catch (err) {
        if (err instanceof DOMException && err.name === 'AbortError') {
          // success — cooperative cancellation
          return;
        }
        await messages.sendMessage('EXPERIMENT_ERROR', {
          id,
          phase: 'cleanup',
          message: String(err),
          stack: err instanceof Error ? err.stack : undefined,
        });
      }
    }

    describe('AbortSignal contract (ENG-04 / D-02 / D-11)', () => {
      beforeEach(() => {
        vi.spyOn(messages, 'sendMessage').mockResolvedValue({ ok: true } as never);
      });

      it('controller.abort() fires BEFORE cleanup() is invoked (signal.aborted === true at cleanup time)', async () => {
        const controller = new AbortController();
        let abortedAtCleanup: boolean | null = null;
        const cleanup = async () => {
          abortedAtCleanup = controller.signal.aborted;
        };
        await runCleanup(cleanup, controller, 'X');
        expect(abortedAtCleanup).toBe(true);
      });

      it('AbortError thrown from cleanup() is NOT escalated via EXPERIMENT_ERROR', async () => {
        const controller = new AbortController();
        const cleanup = async () => {
          throw new DOMException('Aborted', 'AbortError');
        };
        await runCleanup(cleanup, controller, 'X');
        expect(messages.sendMessage).not.toHaveBeenCalled();
      });

      it('non-AbortError thrown from cleanup() IS escalated via EXPERIMENT_ERROR with phase=cleanup', async () => {
        const controller = new AbortController();
        const err = new Error('real cleanup boom');
        const cleanup = async () => {
          throw err;
        };
        await runCleanup(cleanup, controller, '01J0AAAAAAAAAAAAAAAAAAAAAA');
        expect(messages.sendMessage).toHaveBeenCalledWith(
          'EXPERIMENT_ERROR',
          expect.objectContaining({
            id: '01J0AAAAAAAAAAAAAAAAAAAAAA',
            phase: 'cleanup',
            message: expect.stringContaining('real cleanup boom'),
          }),
        );
      });
    });
    ```
