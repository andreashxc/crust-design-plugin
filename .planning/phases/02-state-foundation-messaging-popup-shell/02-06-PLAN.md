    **C) `ExperimentRow.tsx`** (UI-SPEC "Card internal layout" + "Status Badge Contract" + "Switch (Toggle) Behavior Contract" + "Stack-trace Collapsible"):

    ```tsx
    /**
     * Per-experiment Card row.
     *
     * Composition (UI-SPEC layout):
     *  Row 1 (justify-between): name + status Badge.
     *  Row 2 (mt-1): description (line-clamp-2 muted).
     *  Row 3 (mt-2 justify-end): Loader2 (when pending) + Switch.
     *  Conditional (status === 'error' || 'auto-disabled'):
     *    - Auto-disable banner (only for 'auto-disabled').
     *    - Last error 1-line (red).
     *    - Show/Hide stack trace toggle (ghost Button) → Collapsible (mono).
     *  Conditional (toggleErrored): red inline alert.
     *
     * Status badge variant override for 'auto-disabled' (UI-SPEC line 224
     * mandate to keep this comment in source):
     *   The auto-disabled badge is "destructive outline" — there is no built-in
     *   shadcn variant for this. Override solid red with className.
     */
    import { useEffect, useState } from 'react';
    import { AlertCircle, ChevronDown, ChevronRight, Loader2 } from 'lucide-react';
    import { Badge } from '@/components/ui/badge';
    import { Button } from '@/components/ui/button';
    import { Card } from '@/components/ui/card';
    import {
      Collapsible,
      CollapsibleContent,
      CollapsibleTrigger,
    } from '@/components/ui/collapsible';
    import { ScrollArea } from '@/components/ui/scroll-area';
    import { Switch } from '@/components/ui/switch';
    import type { ExperimentStatus, RegistryEntry } from '@platform/experiment-sdk';
    import { sendMessage } from '@/shared/messages';
    import { matchesUrl } from '@/shared/url-match';
    import { computeStatus } from '@/popup/status';
    import { useStore } from '@/popup/store';

    // D-11 constants for the auto-disable banner (mirrored from
    // src/background/handlers/experiment-error.ts; if those constants change,
    // update both places).
    const WINDOW_SEC = 10;

    type StatusBadgeProps = { status: ExperimentStatus };
    function StatusBadge({ status }: StatusBadgeProps) {
      // 5-variant mapping (D-23 verbatim).
      switch (status) {
        case 'disabled':
          return <Badge variant="secondary">Off</Badge>;
        case 'pending':
          return <Badge variant="outline">Pending</Badge>;
        case 'applied':
          return <Badge variant="default">Applied</Badge>;
        case 'error':
          return <Badge variant="destructive">Error</Badge>;
        case 'auto-disabled':
          // UI-SPEC line 224: "destructive outline" via className override.
          return (
            <Badge
              variant="destructive"
              className="bg-transparent text-destructive border-destructive border"
            >
              Auto-disabled
            </Badge>
          );
      }
    }

    function truncateMessage(message: string, max = 80): string {
      return message.length > max ? `${message.slice(0, max)}…` : message;
    }

    export function ExperimentRow({ entry }: { entry: RegistryEntry }) {
      // Subscribed selectors — every row re-renders when its slice of state
      // changes (Blocker 3 fix; the original draft used useStore.getState() for
      // appliedInActiveTab which is a one-shot read and never re-rendered when
      // chrome.storage.onChanged fired).
      const enabled = useStore((s) => s.enabled[entry.id] ?? false);
      const autodisabled = useStore((s) => s.autodisabled[entry.id]);
      const lastError = useStore((s) => s.lastError[entry.id]);
      const appliedInActiveTab = useStore((s) => s.appliedInActiveTab);
      // Blocker 4 fix: activeTabUrl is now read from the store (Plan 02-05's
      // TabSlice + Plan 02-06 main.tsx bootstrap populates it via
      // chrome.tabs.query .url). computeStatus uses this to resolve
      // 'pending' vs 'applied' per CONTEXT D-26 / D-08. The Phase 2 limitation
      // around UI-05 is gone — `Applied` now renders correctly on first popup
      // open.
      const activeTabUrl = useStore((s) => s.activeTabUrl) ?? undefined;

      const status = computeStatus(entry, {
        enabled,
        autodisabled,
        appliedInActiveTab,
        activeTabUrl,
      });

      const [pending, setPending] = useState(false);
      const [toggleError, setToggleError] = useState<string | null>(null);
      const [stackOpen, setStackOpen] = useState(false);

      // Warning 10 fix: clear stale toggleError when storage.onChanged delivers
      // a fresh enabled state (deterministic; doesn't rely on a setTimeout race).
      // Any change in `enabled` from storage = the SW write succeeded → the
      // earlier failure message is no longer relevant.
      useEffect(() => {
        setToggleError(null);
      }, [enabled]);

      async function handleToggle(next: boolean): Promise<void> {
        setPending(true);
        setToggleError(null);
        try {
          const result = await sendMessage('EXPERIMENT_TOGGLE', {
            id: entry.id,
            enabled: next,
          });
          if (!result.ok) {
            setToggleError(result.error);
          }
          // On success, the SW broadcasts STATE_CHANGED; chrome.storage.onChanged
          // updates the store, which re-renders this row with the new enabled
          // state. No optimistic local toggle needed — the Switch's checked
          // value reads from the store selector above.
        } catch (err) {
          setToggleError(String(err));
        } finally {
          setPending(false);
        }
      }

      const showError = status === 'error' || status === 'auto-disabled';

      return (
        <Card className="p-3">
          <div className="flex items-start justify-between gap-2">
            <span className="text-sm font-semibold truncate">{entry.name}</span>
            <StatusBadge status={status} />
          </div>
          <p className="mt-1 text-sm text-muted-foreground line-clamp-2">
            {entry.description}
          </p>
          <div className="mt-2 flex items-center justify-end gap-2">
            {pending ? (
              <Loader2 className="size-3 animate-spin" aria-hidden="true" />
            ) : null}
            <Switch
              aria-label={`Toggle ${entry.name}`}
              checked={enabled}
              disabled={pending}
              onCheckedChange={(next) => {
                void handleToggle(next);
              }}
            />
          </div>
          {showError ? (
            <div className="mt-2">
              {status === 'auto-disabled' && autodisabled ? (
                <p className="flex items-center gap-1 text-xs font-medium text-destructive">
                  <AlertCircle className="size-3" aria-hidden="true" />
                  <span>
                    Disabled after {autodisabled.count} errors in {WINDOW_SEC}s {'—'} toggle
                    off/on to re-enable.
                  </span>
                </p>
              ) : null}
              {lastError ? (
                <>
                  <p className="text-xs text-destructive mt-1">
                    Last error: {truncateMessage(lastError.message, 80)}
                  </p>
                  {lastError.stack ? (
                    <Collapsible open={stackOpen} onOpenChange={setStackOpen}>
                      <CollapsibleTrigger asChild>
                        <Button
                          variant="ghost"
                          size="sm"
                          className="h-auto p-0 text-xs font-medium mt-1"
                        >
                          {stackOpen ? (
                            <ChevronDown className="size-3 mr-1" aria-hidden="true" />
                          ) : (
                            <ChevronRight className="size-3 mr-1" aria-hidden="true" />
                          )}
                          {stackOpen ? 'Hide stack trace' : 'Show stack trace'}
                        </Button>
                      </CollapsibleTrigger>
                      <CollapsibleContent>
                        <ScrollArea className="max-h-[140px] mt-2">
                          <div className="rounded border bg-muted/40 p-2 text-xs font-mono whitespace-pre">
                            {lastError.stack}
                          </div>
                        </ScrollArea>
                      </CollapsibleContent>
                    </Collapsible>
                  ) : null}
                </>
              ) : null}
            </div>
          ) : null}
          {toggleError ? (
            <p className="text-xs text-destructive mt-2">Toggle failed: {toggleError}</p>
          ) : null}
        </Card>
      );
    }
    ```

    ```tsx
    /**
     * Popup mount.
     *
     * Phase 2 evolution:
     *  - Bootstrap fetches registry.json + reads chrome.storage.{local,session}
     *    BEFORE React mounts (RESEARCH §8 verbatim).
     *  - Subscribes to chrome.storage.onChanged at module top so the popup's
     *    Zustand store mirrors storage as the SW writes (D-09 / D-27).
     *  - Mount idiom (createRoot + StrictMode) preserved verbatim from Phase 1.
     */
    import { StrictMode } from 'react';
    import { createRoot } from 'react-dom/client';
    import '@/styles/popup.css';
    import { App } from './App';
    import { useStore } from '@/popup/store';
    import {
      getAppliedInTab,
      getAutoDisabled,
      getEnabledExperiments,
      getLastErrors,
    } from '@/shared/storage';

    async function bootstrapStore(): Promise<void> {
      const [enabled, autodisabled, lastError, registryRes] = await Promise.all([
        getEnabledExperiments(),
        getAutoDisabled(),
        getLastErrors(),
        fetch(chrome.runtime.getURL('registry.json')).then((r) => r.json()),
      ]);
      useStore.setState({ enabled, autodisabled, lastError, registry: registryRes });

      // Blocker 4 fix: capture activeTab.url alongside activeTab.id so
      // computeStatus can resolve 'pending' vs 'applied' on first popup open
      // (CONTEXT D-26: full URL needed for scope match against the live page).
      const [activeTab] = await chrome.tabs.query({ active: true, currentWindow: true });
      if (activeTab?.id != null) {
        const applied = await getAppliedInTab(activeTab.id);
        useStore.setState({
          activeTabId: activeTab.id,
          activeTabUrl: activeTab.url ?? null,
          appliedInActiveTab: applied,
        });
      }
    }

    // Subscribe to storage changes — popup is alive while open; listener is
    // automatically released when the popup closes (storage.onChanged
    // listeners scoped to extension context lifecycle).
    chrome.storage.onChanged.addListener((changes, areaName) => {
      if (areaName === 'local') {
        if (changes.enabled) {
          useStore.setState({ enabled: (changes.enabled.newValue as Record<string, boolean>) ?? {} });
        }
        if (changes.autodisabled) {
          useStore.setState({
            autodisabled:
              (changes.autodisabled.newValue as Record<string, import('@platform/experiment-sdk').AutoDisableRecord>) ??
              {},
          });
        }
        if (changes.last_error) {
          useStore.setState({
            lastError:
              (changes.last_error.newValue as Record<string, import('@platform/experiment-sdk').ErrorRecord>) ??
              {},
          });
        }
      } else if (areaName === 'session') {
        const tabId = useStore.getState().activeTabId;
        if (tabId != null) {
          const key = `applied:${tabId}`;
          if (changes[key]) {
            useStore.setState({ appliedInActiveTab: (changes[key].newValue as string[]) ?? [] });
          }
        }
      }
    });

    void bootstrapStore();

    const container = document.getElementById('root');
    if (!container) throw new Error('popup: #root missing');
    createRoot(container).render(
      <StrictMode>
        <App />
      </StrictMode>,
    );
    ```
