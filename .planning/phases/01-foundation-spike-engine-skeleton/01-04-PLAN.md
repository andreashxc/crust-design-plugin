```

The plugin's `buildStart` hook does FOUR things in order, for every manifest under `experiments/*/*/manifest.json`:
1. Parse the JSON file.
2. If `id` is missing or empty: generate a ULID via the `ulid` package, write it back to the file (preserving formatting), log `Wrote new id to <path> — please commit`.
3. Run `ExperimentManifest.safeParse(parsed)`. On failure, format every Zod issue and `throw new Error(formatted)` — Vite turns this into a build failure.
4. Verify `parsed.author === <author>` where `<author>` is the directory segment under `experiments/`. Mismatch throws a build error.

The plugin does NOT emit `dist/registry.json` (that is BLD-03 in Phase 2). Runtime discovery in content scripts happens via `import.meta.glob` (Plan 03).

The plugin does NOT need a Vite alias — the WXT config (Plan 03) already declares `@experiments` for runtime resolution.

    /**
     * Vite plugin: discover, validate, and ULID-stamp experiments at build time.
     *
     * Per CONTEXT D-17 (ULID auto-gen), D-18 (author == folder), D-20 (discovery),
     *      D-21 (Zod validation), D-22 (no registry.json — Phase 2).
     * Per BLD-01 + BLD-02 + MAN-01.
     */

    import type { Plugin } from 'vite';
    import { globSync } from 'glob';
    import { readFileSync, writeFileSync } from 'node:fs';
    import { dirname, relative, resolve, sep } from 'node:path';
    import { ulid } from 'ulid';
    import { ExperimentManifest } from '@platform/experiment-sdk';

    export type BuildExperimentsOptions = {
      /** Repo root containing the `experiments/` folder. Defaults to `process.cwd()`. */
      root?: string;
    };

    export function buildExperiments(options: BuildExperimentsOptions = {}): Plugin {
      const root = options.root ?? process.cwd();
      return {
        name: 'platform:build-experiments',
        buildStart() {
          const result = scanAndValidate(root);
          if (result.errors.length > 0) {
            throw new Error(formatErrors(result.errors));
          }
          // result.warnings are informational (e.g., ULID written) — already logged.
        },
      };
    }

    // ---- Pure helpers (exported for tests) ----

    export type ScanResult = {
      manifests: Array<{ path: string; data: ReturnType<typeof ExperimentManifest.parse> }>;
      errors: BuildExperimentError[];
      warnings: string[];
    };

    export type BuildExperimentError = {
      file: string;
      kind: 'parse' | 'schema' | 'author-mismatch';
      issues: Array<{ path: string; message: string }>;
    };

    export function scanAndValidate(root: string): ScanResult {
      // Discover all `experiments/<author>/<id>/manifest.json` — depth exactly 2.
      const matches = globSync('experiments/*/*/manifest.json', { cwd: root, absolute: true });

      const manifests: ScanResult['manifests'] = [];
      const errors: BuildExperimentError[] = [];
      const warnings: string[] = [];

      for (const absPath of matches) {
        const fileRel = relative(root, absPath);
        let raw: string;
        let parsedJson: unknown;
        try {
          raw = readFileSync(absPath, 'utf8');
          parsedJson = JSON.parse(raw);
        } catch (err) {
          errors.push({
            file: fileRel,
            kind: 'parse',
            issues: [{ path: '<root>', message: `JSON parse failed: ${String(err)}` }],
          });
          continue;
        }

        // ULID write-back (D-17). Only when id is missing OR empty string.
        if (
          parsedJson &&
          typeof parsedJson === 'object' &&
          (!('id' in parsedJson) || (parsedJson as { id?: unknown }).id === '' || (parsedJson as { id?: unknown }).id == null)
        ) {
          const newId = ulid();
          (parsedJson as { id: string }).id = newId;
          // Preserve the original formatting heuristic: 2-space indent + trailing newline.
          const out = `${JSON.stringify(parsedJson, null, 2)}\n`;
          writeFileSync(absPath, out, 'utf8');
          const msg = `[build-experiments] Wrote new id to ${fileRel} — please commit`;
          // eslint-disable-next-line no-console
          console.log(msg);
          warnings.push(msg);
        }

        // Zod validation (D-21 / BLD-02).
        const parsed = ExperimentManifest.safeParse(parsedJson);
        if (!parsed.success) {
          errors.push({
            file: fileRel,
            kind: 'schema',
            issues: parsed.error.issues.map((issue) => ({
              path: issue.path.length === 0 ? '<root>' : issue.path.join('.'),
              message: issue.message,
            })),
          });
          continue;
        }

        // Author-vs-folder check (D-18).
        const expectedAuthor = authorFromPath(fileRel);
        if (expectedAuthor !== parsed.data.author) {
          errors.push({
            file: fileRel,
            kind: 'author-mismatch',
            issues: [
              {
                path: 'author',
                message: `manifest.author "${parsed.data.author}" must match folder name "${expectedAuthor}"`,
              },
            ],
          });
          continue;
        }

        manifests.push({ path: fileRel, data: parsed.data });
      }

      return { manifests, errors, warnings };
    }

    /**
     * Given `experiments/andrew/smoke/manifest.json`, returns `andrew`.
     * Works on POSIX and Windows path separators.
     */
    export function authorFromPath(fileRel: string): string {
      const parts = fileRel.split(sep).filter(Boolean);
      // Expect: ['experiments', '<author>', '<id>', 'manifest.json']
      if (parts.length !== 4 || parts[0] !== 'experiments' || parts[3] !== 'manifest.json') {
        return '<unknown>';
      }
      return parts[1] ?? '<unknown>';
    }

    export function formatErrors(errors: BuildExperimentError[]): string {
      const lines: string[] = ['Manifest validation failed:'];
      for (const e of errors) {
        lines.push(`  ${e.file}`);
        for (const issue of e.issues) {
          lines.push(`    - ${issue.path}: ${issue.message}`);
        }
      }
      return lines.join('\n');
    }

    /** Resolve repo root from a known marker (pnpm-workspace.yaml). Used by tests. */
    export function findRepoRoot(startDir: string): string {
      let dir = startDir;
      for (let i = 0; i < 10; i++) {
        try {
          readFileSync(resolve(dir, 'pnpm-workspace.yaml'), 'utf8');
          return dir;
        } catch {
          const parent = dirname(dir);
          if (parent === dir) break;
          dir = parent;
        }
      }
      return startDir;
    }
    ```

    import { mkdirSync, rmSync, writeFileSync, readFileSync, existsSync } from 'node:fs';
    import { resolve } from 'node:path';
    import { tmpdir } from 'node:os';
    import {
      scanAndValidate,

    beforeEach(() => {

    afterEach(() => {
      rmSync(tmpRoot, { recursive: true, force: true });

    function writeManifest(rel: string, value: unknown): string {
      const abs = resolve(tmpRoot, rel);
      mkdirSync(resolve(abs, '..'), { recursive: true });
      writeFileSync(abs, `${JSON.stringify(value, null, 2)}\n`, 'utf8');
      return abs;
    }

    function writeFixture(rel: string, fixtureName: string): string {
      // Fixtures live at <repo>/tests/fixtures/manifests/<name>.json. Find repo from CWD.

      const raw = readFileSync(fixturePath, 'utf8');

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| filesystem (designer commits) → build pipeline | A designer's manifest.json crosses this boundary at every build |
| build pipeline → file write-back | The plugin writes back to `manifest.json` to stamp ULIDs — this is a privileged operation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-01 | Tampering / DoS | malformed manifest.json crashes the build | mitigate | The plugin's `scanAndValidate` returns structured errors instead of throwing inside `JSON.parse`; only the orchestrator-level `formatErrors` + `throw new Error` triggers Vite's build failure path with a readable message. The 12+ tests prove every Zod rule from D-16 surfaces via this code path. BLD-02. |
| T-1-06 | Spoofing | manifest claims `author: "alice"` while sitting under `experiments/bob/` | mitigate | `scanAndValidate` checks `parsed.author === authorFromPath(fileRel)` and produces an `author-mismatch` error. Tested by the "throws when manifest.author does not match the folder name" case. D-18. |
| T-1-WB | Tampering | the plugin writes back to the manifest file (ULID generation) | accept | Only triggered when `id` is empty/missing. Idempotent (second run does not re-write). Authored manifests with a non-empty id are NEVER modified. The write is logged so a designer notices the diff. D-17. |
</threat_model>
